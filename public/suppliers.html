<body>

  <h1>Find Sustainable Suppliers</h1>

  <label>Material Category: <input id="material" placeholder="e.g. cotton"></label><br>

  <label>Max Lead Time (days): <input id="lead" type="number" placeholder="e.g. 10"></label><br>

  <label>Min ESG Score: <input id="esg" type="number" placeholder="e.g. 70"></label><br>

  <button onclick="findSuppliers()">Search</button>


  <table id="results" style="display:none;">

    <thead>

      <tr>

        <th>Name</th><th>City</th><th>Country</th><th>Material</th><th>Cost (£)</th>

        <th>Lead Time (days)</th><th>MOQ</th><th>ESG Score</th><th>Certifications</th>

      </tr>

    </thead>

    <tbody></tbody>

  </table>


  <script>

    async function findSuppliers() {

      const mat = document.getElementById('material').value;

      const lead = document.getElementById('lead').value;

      const esg = document.getElementById('esg').value;

      const res = await fetch(`/api/find_suppliers?material_category=${encodeURIComponent(mat)}&max_lead_days=${lead}&min_esg=${esg}`);

      const data = await res.json();

      const tbody = document.querySelector('#results tbody');

      tbody.innerHTML = "";

      data.suppliers.forEach(s => {

        const row = `<tr>

          <td>${s.supplier_name}</td>

          <td>${s.city}</td>

          <td>${s.country}</td>

          <td>${s.material_category}</td>

          <td>${s.unit_cost_gbp}</td>

          <td>${s.lead_time_days}</td>

          <td>${s.moq_units}</td>

          <td>${s.esg_score_0_100}</td>

          <td>${s.certifications}</td>

        </tr>`;

        tbody.innerHTML += row;

      });

      document.getElementById('results').style.display = data.suppliers.length ? "table" : "none";

    }

  </script>

</body>
