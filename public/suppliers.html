  <!-- Results table -->

  <table id="results" style="display:none;">

      <tr>

  <script>

    const form = document.getElementById('search-form');

    const table = document.getElementById('results');

    const tbody = table.querySelector('tbody');

    const statusEl = document.getElementById('status');


    form.addEventListener('submit', async (e) => {

      e.preventDefault();

      statusEl.textContent = 'Searching…';

      tbody.innerHTML = '';

      table.style.display = 'none';


      const mat = document.getElementById('material').value.trim();

      const lead = document.getElementById('lead').value.trim();

      const esg  = document.getElementById('esg').value.trim();


      const url = `/api/find_suppliers?material_category=${encodeURIComponent(mat)}`

        + (lead ? `&max_lead_days=${encodeURIComponent(lead)}` : '')

        + (esg  ? `&min_esg=${encodeURIComponent(esg)}` : '');


      try {

        const res = await fetch(url, { headers: { 'Accept': 'application/json' } });

        if (!res.ok) throw new Error(`HTTP ${res.status}`);

        const data = await res.json();


        if (!Array.isArray(data) || data.length === 0) {

          statusEl.textContent = 'No results found.';

          return;

        }


        data.forEach(s => {

          const tr = document.createElement('tr');

          tr.innerHTML = `

            <td>${s.name ?? ''}</td>

            <td>${s.material ?? ''}</td>

            <td>${s.lead_days ?? ''}</td>

            <td>${s.esg_score ?? ''}</td>`;

          tbody.appendChild(tr);

        });


        table.style.display = 'table';

        statusEl.textContent = '';

      } catch (err) {

        statusEl.textContent = 'Error fetching suppliers. Check the API route.';

        console.error(err);

      }

    });

  </script>
